# Grammer Language Syntax
- * : none or more
- | : or
- , : and
- .. : spread
- ; : end of line

# Op Language Synatx 
- Ident = letter , char*;

- Keyword = ForKeyword | IfKeyword | WhileKeyword;
- ForKeyword = "for";
- IfKeyword = "if";
- WhileKeyword = "while";

- String = " , char* , "; 
- Number = digit | digit  , . , digit*;
- ArhOp = "+" | "-" | "*" | "/";
- BooleanOp = "&&" | "||" | ">" | "<" | ">=" | "<=" | "==";
- SemiCol = ";";
- Equal = "=";
- digit = 0 | .. | 9;
- char = letter | digit;
- letter = a | .. | z | A | .. | Z;
